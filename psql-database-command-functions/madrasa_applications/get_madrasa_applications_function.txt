

CREATE OR REPLACE FUNCTION public.get_madrasa_applications(
    p_page INTEGER DEFAULT 1,
    p_limit INTEGER DEFAULT 10,
    p_search_term TEXT DEFAULT NULL,
    p_status_filter TEXT DEFAULT NULL
)
RETURNS JSONB
LANGUAGE plpgsql
STABLE
SECURITY DEFINER
AS $$
DECLARE
    _offset INTEGER;
    _items_jsonb JSONB;
    _total_items INTEGER;
    _base_query TEXT;
    _data_query TEXT;
    _count_query TEXT;
    _where_clauses TEXT[] := ARRAY['TRUE'];
BEGIN
    _offset := (p_page - 1) * p_limit;

    IF p_status_filter IS NOT NULL AND p_status_filter <> '' THEN
        _where_clauses := array_append(_where_clauses, format('ma.application_status = %L', p_status_filter));
    END IF;

    IF p_search_term IS NOT NULL AND TRIM(p_search_term) <> '' THEN
        _where_clauses := array_append(_where_clauses, format(
            '(ma.name_bn ILIKE %1$L OR ma.name_ar ILIKE %1$L OR ma.contact_email ILIKE %1$L OR (ma.muhtamim->>''name'') ILIKE %1$L)',
            '%' || TRIM(p_search_term) || '%'
        ));
    END IF;

    _base_query := '
        FROM public.madrasa_applications ma
        LEFT JOIN public.zones z ON ma.zone_id = z.id
        LEFT JOIN public.marhalas m_boys ON ma.highest_marhala_boys_id = m_boys.id
        LEFT JOIN public.marhalas m_girls ON ma.highest_marhala_girls_id = m_girls.id
        WHERE ' || array_to_string(_where_clauses, ' AND ');

    _count_query := 'SELECT COUNT(ma.id)' || _base_query;
    EXECUTE _count_query INTO _total_items;

    _data_query := '
        SELECT 
            ma.*,
            z.name_bn as zone_name_bn,
            m_boys.name_bn as highest_marhala_boys_name_bn,
            m_girls.name_bn as highest_marhala_girls_name_bn
    ' || _base_query || '
        ORDER BY ma.created_at DESC
        LIMIT ' || p_limit || ' OFFSET ' || _offset;
    
    EXECUTE 'SELECT COALESCE(jsonb_agg(row_to_json(q)), ''[]''::jsonb) FROM (' || _data_query || ') q' INTO _items_jsonb;
    
    RETURN jsonb_build_object('items', _items_jsonb, 'totalItems', _total_items);
END;
$$;
GRANT EXECUTE ON FUNCTION public.get_madrasa_applications(INTEGER, INTEGER, TEXT, TEXT) TO authenticated;
